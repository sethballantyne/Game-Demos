class MainForm
Private
	-- mouse cursor when the eraser tool is selected.
	Cursor eraserCursor

	function MainForm_Close()
		EditorLogic->CleanUp()
	end function

	function MainForm_Closing()
		e->Cancel := EditorLogic->ApplicationClosing()
	end function

	function MainForm_Load()
		try
			eraserCursor := new Cursor(GetType(), "Resources.Eraser.cur")

			EditorLogic->Initialize(pictureBox, brickImageList, listView)
			EditorLogic->SetUndoGUIItems(/* pass undo menus */)
			EditorLogic->SetRedoGUIItems(/* pass redo menus */)
		catch
			throw
	end function

	function ListView_SelectedIndexChanged()
		if(listView->SelectedItems->Count > 0)
			EditorLogic->SetEditingMode(EditingMode::Brush)
			EditorLogic->SetSelectedBrick(listView->Items->IndexOf(listView->SelectedItems[0]))
		end if
	end function
end class
